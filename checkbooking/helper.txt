  function loaddata(){
            $.ajax({
                type:"get",
                url:'api/newduty.php',
                data:{},
                success:function(resp)
                {   
                    var jsonData=JSON.parse(resp);
                    if(jsonData.success=='1')
                    {
                        setTimeout(function(){
                            var result = jsonData.result;
                            var html = '';
                            
                            for(var i=0;i < result.length; i++) {
                                html += '<tr><td>'+result[i].bookingid+'</td>'
                                        +'<td>'+result[i].reportdate+'('+result[i].reporttime+')</td>'
                                        +'<td>'+result[i].dutytype+'</td>'    
                                        +'<td>'+result[i].name+'</td>'     
                                        +'<td>'+result[i].contact+'</td>'    
                                        +'<td>'+result[i].location+'</td>'    
                                        +'<td>'+result[i].car+'</td>' 
                                        +'<td> <i class="fa fa-check-circle" style="color:green" id="accept'+result[i].bookingid+'"></i>'
                                        +'&nbsp;&nbsp;&nbsp;&nbsp;|&nbsp;&nbsp;&nbsp;&nbsp;'
                                        +'<i class="fa fa-times-circle" style="color:red" id="'+result[i].bookingid+'"></i></td></tr>'
                                                                                    
                                
                            }
                        },9000);

                        $('#newduty').html(html);
                    }
                }
            })
    
    };
